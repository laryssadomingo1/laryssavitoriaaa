function converter() { // Obtém os valores do input e do select const valor = parseFloat(document.getElementById('valor').value); const tipoConversao = document.getElementById('conversao').value; let resultado; // Verifica se o valor inserido é um número válido if (isNaN(valor)) { document.getElementById('resultadoConv').innerText = 'Por favor, insira um valor válido.'; return; } // Realiza a conversão de acordo com a opção selecionada switch (tipoConversao) { case 'cf': // Conversão de Celsius para Fahrenheit resultado = (valor * 9/5) + 32; document.getElementById('resultadoConv').innerText = resultado.toFixed(2) + ' °F'; break; case 'mc': // Conversão de Metros para Centímetros resultado = valor * 100; document.getElementById('resultadoConv').innerText = resultado + ' cm'; break; case 'kl': // Conversão de Quilos para Libras resultado = valor * 2.20462; document.getElementById('resultadoConv').innerText = resultado.toFixed(2) + ' lbs'; break; default: document.getElementById('resultadoConv').innerText = 'Selecione uma conversão válida.'; break; } }
